`$ git clone git@github.com:leon-do/lnd-static-channel-backup-simnet-experiment.git`

`$ cd static-chan-backup`

install btcd https://github.com/btcsuite/btcd

`btcd$ btcd --txindex --simnet --rpcuser=kek --rpcpass=kek`

start alice's lnd

`alice$ ./lnd-debug --rpclisten=localhost:10001 --listen=localhost:10011 --restlisten=localhost:8001 --datadir=./alice --bitcoin.simnet --bitcoin.active --bitcoin.node=btcd --btcd.rpcuser=kek --btcd.rpcpass=kek --configfile=./alice`

create alice's wallet

`alice$ ./lncli-debug --rpcserver=localhost:10001 create`


`btcd$ btcd --simnet --txindex --rpcuser=kek --rpcpass=kek --miningaddr=rssZ8WH5N9jJvRdCHjkDwVfTDKo5GZdvK3`

mine bitcoin

`alice$ btcctl --simnet --rpcuser=kek --rpcpass=kek generate 400`
